def drawSpaceShip(x, y):
    ship.clear()
    deactivateKeys()
    ship.penup()
    ship.goto(x, y)
    ship.setheading(0)
    ship.pendown()
    ship.fillcolor("black")
    ship.begin_fill()
    # make the rectangle
    ship.forward(20)
    ship.right(90)
    ship.forward(10)
    ship.right(90)
    ship.forward(20)
    ship.right(90)
    ship.forward(10)
    # make the cannon
    ship.right(90)
    ship.forward(5)
    ship.left(90)
    ship.forward(15)
    ship.right(90)
    ship.forward(10)
    ship.right(90)
    ship.forward(15)

    ship.end_fill()
    ship.penup()
    ship.goto(x, y)
    activateKeys()

def drawCloud(x, y):
    c = turtle.Turtle()

    def filled_circle(radius, color):
        c.color(color, color)
        c.begin_fill()
        c.circle(radius)
        c.end_fill()

    radius = 50
    cloud_color = "white"
    c.penup()
    c.goto(x,y)
    c.pendown()
    filled_circle(radius, cloud_color)
    c.forward(radius)
    filled_circle(radius, cloud_color)
    c.right(90)
    filled_circle(radius, cloud_color)
    c.right(90)
    filled_circle(radius, cloud_color)
    c.right(90)
    filled_circle(radius, cloud_color)
    c.right(90)
