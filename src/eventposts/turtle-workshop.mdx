def drawSpaceShip(x, y):\n
    ship.clear()\n
    deactivateKeys()\n
    ship.penup()\n
    ship.goto(x, y)\n
    ship.setheading(0)\n
    ship.pendown()\n
    ship.fillcolor("black")\n
    ship.begin_fill()\n
    # make the rectangle\n
    ship.forward(20)\n
    ship.right(90)\n
    ship.forward(10)\n
    ship.right(90)\n
    ship.forward(20)\n
    ship.right(90)\n
    ship.forward(10)\n
    # make the cannon\n
    ship.right(90)\n
    ship.forward(5)\n
    ship.left(90)\n
    ship.forward(15)\n
    ship.right(90)\n
    ship.forward(10)\n
    ship.right(90)\n
    ship.forward(15)\n\n
    ship.end_fill()\n
    ship.penup()\n
    ship.goto(x, y)\n
    activateKeys()\n\n

def drawCloud(x, y):\n
    c = turtle.Turtle()\n\n
    def filled_circle(radius, color):\n
        c.color(color, color)\n
        c.begin_fill()\n
        c.circle(radius)\n
        c.end_fill()\n\n
    radius = 50\n
    cloud_color = "white"\n
    c.penup()\n
    c.goto(x,y)\n
    c.pendown()\n
    filled_circle(radius, cloud_color)\n
    c.forward(radius)\n
    filled_circle(radius, cloud_color)\n
    c.right(90)\n
    filled_circle(radius, cloud_color)\n
    c.right(90)\n
    filled_circle(radius, cloud_color)\n
    c.right(90)\n
    filled_circle(radius, cloud_color)\n
    c.right(90)\n\n
